Raphael.fn.star = function (cx, cy, r, r2, rays) { r2 = r2 || r * .382; rays = rays || 5; var points = [“M”, cx, cy + r2, “L”],R; for (var i = 1; i < rays * 2; i++) { R = i % 2 ? r : r2; points = points.concat([(cx + R * Math.sin(i * Math.PI / rays)), (cy + R * Math.cos(i * Math.PI / rays))]); } points.push(“z”); return this.path(points.join()); };
